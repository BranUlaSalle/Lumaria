	\section{Descripción del Proyecto}
	
	\subsection{Concepto General}
	
	Soul Knight Fan Remake es un videojuego 2D basado en el popular Soul Knight original. A diferencia del juego original diseñado para móviles, esta versión está optimizada exclusivamente para computadoras y laptops, con controles intuitivos de teclado y mouse.
	
	\subsection{Características Principales}
	
	\begin{itemize}
		\item \textbf{Skins divertidas:} Variedad de personajes con diseños creativos
		\item \textbf{Mapa sencillo e intuitivo:} Diseño de niveles fácil de navegar
		\item \textbf{Modo multijugador local:} Cooperación entre dos jugadores en la misma máquina
		\item \textbf{Sistema de habitaciones:} Cada nivel consta de 5 habitaciones conectadas
		\item \textbf{5 niveles totales:} Progresión con dificultad incremental
		\item \textbf{Jefe final:} Boss con alta cantidad de vida en el nivel 5
	\end{itemize}
	
	\subsection{Entorno del Juego}
	
	La historia del juego es directa y enfocada en la acción:
	
	\begin{enumerate}
		\item El personaje jugable aparece en una habitación inicial equipado con un arma
		\item Las habitaciones están conectadas por caminos
		\item Al entrar a una nueva habitación, las puertas se cierran automáticamente
		\item Aparecen enemigos que el jugador debe derrotar
		\item Una vez eliminados todos los enemigos, las puertas se abren
		\item Cada nivel tiene 5 habitaciones
		\item El juego cuenta con 5 niveles en total
		\item En el nivel 5 (última habitación) aparece un jefe final con mucha vida
	\end{enumerate}
	
	\subsection{Elementos del Juego}
	
	\subsubsection{Items principales}
	\begin{itemize}
		\item Personajes jugables
		\item Enemigos variados
		\item Jefe final (boss)
		\item Mapas/Habitaciones
		\item Objetos consumibles (curas o bebidas)
		\item Variedad de armas
	\end{itemize}

	\end{itemize}

	\subsubsection{Archivo: MiniDungeon.java (Versión Completa)}
	
	\begin{lstlisting}[language=Java, caption={Clase Principal del Juego - MiniDungeon.java}]
import java.util.Random;
import java.util.Scanner;

// Clase base para el personaje jugador
class Jugador {
    int vida = 20;
    int dano = 5;
    int x = 0, y = 0;
}

// Clase para el enemigo
class Enemigo {
    int vida = 10;
    int dano = 3;
    int x, y;
    
    public Enemigo(int x, int y) {
        this.x = x;
        this.y = y;
    }
}

// Clase principal del juego
public class MiniDungeon {
    static final int TAM = 5;
    static Jugador jugador = new Jugador();
    static Enemigo enemigo = new Enemigo(3, 3);
    static Scanner sc = new Scanner(System.in);
    
    public static void main(String[] args) {
        System.out.println("=== MINI DUNGEON CONSOLE ===");
        System.out.println("Controles: w=arriba, s=abajo, a=izquierda, d=derecha, x=atacar\n");
        
        // Bucle principal del juego
        while (jugador.vida > 0 && enemigo.vida > 0) {
            mostrarMapa();
            System.out.print("Mover > ");
            char mov = sc.next().charAt(0);
            moverJugador(mov);
            moverEnemigo();
            
            if (enemigo.vida <= 0) break;
        }
        
        // Resultado final
        if (jugador.vida <= 0) 
            System.out.println("\n�� Has sido derrotado...");
        else 
            System.out.println("\n�� ¡Ganaste la batalla!");
        
        sc.close();
    }
    
    // Muestra el mapa en la consola
    static void mostrarMapa() {
        System.out.println("\n--- MAPA ---");
        for (int i = 0; i < TAM; i++) {
            for (int j = 0; j < TAM; j++) {
                if (jugador.x == i && jugador.y == j)
                    System.out.print("P "); // Jugador
                else if (enemigo.x == i && enemigo.y == j)
                    System.out.print("E "); // Enemigo
                else
                    System.out.print(". "); // Espacio vacio
            }
            System.out.println();
        }
        System.out.println("❤️ Vida Jugador: " + jugador.vida + 
                         " | �� Vida Enemigo: " + enemigo.vida);
    }
    
    // Controla el movimiento y ataque del jugador
    // By Carlos Peralta
    static void moverJugador(char mov) {
        switch (mov) {
            case 'w': 
                if (jugador.x > 0) jugador.x--; 
                break;
            case 's': 
                if (jugador.x < TAM - 1) jugador.x++; 
                break;
            case 'a': 
                if (jugador.y > 0) jugador.y--; 
                break;
            case 'd': 
                if (jugador.y < TAM - 1) jugador.y++; 
                break;
            case 'x':
                // Sistema de ataque
                if (jugador.x == enemigo.x && jugador.y == enemigo.y) {
                    enemigo.vida -= jugador.dano;
                    System.out.println("⚔️ Buen ataque (-" + jugador.dano + " HP)");
                } else {
                    System.out.println("❌ No hay enemigo aqui");
                }
                return;
        }
    }
    
    // Controla el movimiento aleatorio del enemigo
    static void moverEnemigo() {
        Random r = new Random();
        int mov = r.nextInt(4);
        
        switch (mov) {
            case 0: if (enemigo.x > 0) enemigo.x--; break;
            case 1: if (enemigo.x < TAM - 1) enemigo.x++; break;
            case 2: if (enemigo.y > 0) enemigo.y--; break;
            case 3: if (enemigo.y < TAM - 1) enemigo.y++; break;
        }
        
        // Si el enemigo alcanza al jugador, lo ataca
        if (enemigo.x == jugador.x && enemigo.y == jugador.y) {
            jugador.vida -= enemigo.dano;
            System.out.println("�� ¡Te atacan! (-" + enemigo.dano + " HP)");
        }
    }
}
	\end{lstlisting}
	
	\subsection{Explicación del Código}
	
	\subsubsection{Clase Jugador}
	Define las propiedades básicas del personaje jugable:
	\begin{itemize}
		\item \texttt{vida}: Puntos de vida del jugador (inicia con 20)
		\item \texttt{dano}: Daño que inflige al atacar (5 puntos)
		\item \texttt{x, y}: Posición en el mapa (inicia en 0,0 - esquina superior izquierda)
	\end{itemize}
	
	\subsubsection{Clase Enemigo}
	Define las propiedades del enemigo:
	\begin{itemize}
		\item \texttt{vida}: Puntos de vida del enemigo (inicia con 10)
		\item \texttt{dano}: Daño que inflige al jugador (3 puntos)
		\item \texttt{x, y}: Posición inicial (3,3 - cerca del centro del mapa)
		\item Constructor que permite definir la posición inicial
	\end{itemize}
	
	\subsubsection{Método main()}
	Contiene el flujo principal del juego:
	\begin{enumerate}
		\item Muestra el título y controles
		\item Inicia el bucle principal que se ejecuta mientras ambos estén vivos
		\item En cada iteración:
		\begin{itemize}
			\item Muestra el mapa actual
			\item Lee el input del jugador
			\item Mueve al jugador según el input
			\item Mueve al enemigo aleatoriamente
		\end{itemize}
		\item Al finalizar, muestra el resultado (victoria o derrota)
	\end{enumerate}
	
	\subsubsection{Método mostrarMapa()}
	Dibuja el estado actual del juego en la consola:
	\begin{itemize}
		\item Recorre una matriz de 5x5
		\item Muestra "P" donde está el jugador
		\item Muestra "E" donde está el enemigo
		\item Muestra "." en espacios vacíos
		\item Muestra las vidas actuales de ambos
	\end{itemize}
	
	\subsubsection{Método moverJugador()}
	   Procesa el input del jugador:
	\begin{itemize}
		\item \texttt{w}: Mueve arriba (decrementa x)
		\item \texttt{s}: Mueve abajo (incrementa x)
		\item \texttt{a}: Mueve izquierda (decrementa y)
		\item \texttt{d}: Mueve derecha (incrementa y)
		\item \texttt{x}: Ataca si está en la misma posición que el enemigo
		\item Valida que no se salga de los límites del mapa (0 a TAM-1)
	\end{itemize}
	
	\subsubsection{Método moverEnemigo()}
	Controla el comportamiento del enemigo:
	\begin{itemize}
		\item Genera un número aleatorio entre 0 y 3
		\item Mueve al enemigo en una dirección aleatoria
		\item Valida límites del mapa
		\item Si está en la misma posición que el jugador, lo ataca automáticamente
	\end{itemize}
